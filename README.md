#### HOLESKY NETWORK

```
  const { chains, provider } = configureChains(
  [
    {
      id: 17000,
      name: "Holesky",
      network: "holesky",
      nativeCurrency: {
        name: "Holesky Ether",
        symbol: "ETH",
        decimals: 18,
      },
      rpcUrls: {
        default: {
          http: [`${HOLESKY}`],
        },
        public: {
          http: [`${HOLESKY}`],
        },
      },
      blockExplorers: {
        default: {
          name: "Holescan",
          url: "https://holesky.etherscan.io/",
        },
      },
      testnet: true,
    },
  ],
  [
    jsonRpcProvider({
      rpc: (chain) => {
        if (chain.id === 17000) {
          return { http: `${HOLESKY}` };
        }
        return null;
      },
      priority: 1,
    }),
  ]
);
```
